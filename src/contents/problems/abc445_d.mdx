## 問い

$W×H$ ブロックの板チョコを手に持っている。\
これをブロックの境界に沿って $2$ 分割し、片方を机に置く。\
この操作を $N-1$ 回繰り返したあと、最後に手に持っているピースも机に置いた。\
できた $N$ 枚のピースが元の向きで与えられるので、\
$W×H$ ブロックの板チョコになるよう配置を復元せよ。

## 解く

- [1分悩んでダメだったので解説を見に行った](https://atcoder.jp/contests/abc445/editorial/15896)
- ある分割時点で手に持っているピースの大きさを $w×h$ とすると、\
  少なくとも分割後机に置いたほうのピースは、幅と高さのどちらかが $w$ か $h$ のまま
- つまり、初期時点で $w=W,h=H$ として、幅か高さが $w$ か $h$ に一致する\
  ピースを追いかけて $w,h$ を小さくしていけば、自動的に全ての位置が決まる

## 書く

イテレータゴリ押し実装はやめたほうがいい

```cpp
void solve() {
    ll cin(H,W,n);
    vtll3 HWI;
    rep(i,n){ll cin(h,w);HWI.emplace_back(h,w,i);}
    set<ll> used;
    auto HWI2=HWI;
    sort(all(HWI),comp<-1>); //H降順ソート
    sort(all(HWI2),comp<-2>); //W降順ソート
    auto itr=HWI.begin();
    auto itr2=HWI2.begin();
    vpll ans(n);
    ll r=0,c=0;
    rep(j,n){
        ll h=-1,w=-1;
        if(itr!=HWI.end())h=get<0>(*itr);
        if(itr2!=HWI2.end())w=get<1>(*itr2);
        if(h==H-r){
            auto i=get<2>(*itr);
            ans[i]={r,c};
            used.insert(i);
            c+=get<1>(*itr);
            itr++;
        }
        else if(w==W-c){
            auto i=get<2>(*itr2);
            ans[i]={r,c};
            used.insert(i);
            r+=get<0>(*itr2);
            itr2++;
        }
        while(itr!=HWI.end()&&used.contains(get<2>(*itr)))itr++;
        while(itr2!=HWI2.end()&&used.contains(get<2>(*itr2)))itr2++;
    }
    for(auto&[r,c]:ans)cout<<r+1<<sp<<c+1<<el;
}
```

void solve() {
    ll cin(h,w,n);
    vll H(n),W(n);
    pqueue<pll,Comp<1>> pq_h,pq_w;
    rep(i,n){
        cin>>H[i]>>W[i];
        pq_h.emplace(H[i],i);
        pq_w.emplace(W[i],i);
    }
    vpll ans(n);
    vbool used(n);
    ll r=0,c=0;
    rep(j,n){
        while(pq_h.size()>0&&used[pq_h.top().second])pq_h.pop();
        while(pq_w.size()>0&&used[pq_w.top().second])pq_w.pop();
        if(pq_h.size()>0&&pq_h.top().first==h-r){
            auto i=pq_h.top().second;
            ans[i]={r,c};
            used[i]=true;
            c+=W[i];
        }
        else{
            auto i=pq_w.top().second;
            ans[i]={r,c};
            used[i]=true;
            r+=H[i];
        }
    }
    for(auto&[r,c]:ans)cout<<r+1<<sp<<c+1<<el;
}
